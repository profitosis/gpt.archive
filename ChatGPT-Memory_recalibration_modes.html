# qpmq_core.py

import numpy as np
from sentence_transformers import SentenceTransformer
from queue import PriorityQueue
import uuid, time

class QuantizedPromptMemoryQueue:
    def __init__(self, model_name='all-MiniLM-L6-v2', quant_bits=8):
        self.encoder = SentenceTransformer(model_name)
        self.quant_bits = quant_bits
        self.q = PriorityQueue()
        self.prompt_db = {}  # maps prompt_id to full prompt and metadata

    def _quantize(self, vec):
        scale = 2 ** self.quant_bits - 1
        min_val, max_val = np.min(vec), np.max(vec)
        return ((vec - min_val) / (max_val - min_val + 1e-8) * scale).astype(np.uint8)

    def add_prompt(self, prompt, importance_score=1.0):
        vec = self.encoder.encode(prompt)
        qvec = self._quantize(vec)
        prompt_id = str(uuid.uuid4())
        timestamp = time.time()

        self.prompt_db[prompt_id] = {
            "prompt": prompt,
            "vector": qvec.tolist(),
            "importance": importance_score,
            "timestamp": timestamp
        }

        # Priority = -importance_score + entropy injection
        priority = -(importance_score + np.std(qvec) / 10.0)
        self.q.put((priority, prompt_id))

    def retrieve_top(self, k=5):
        top_prompts = []
        for _ in range(min(k, self.q.qsize())):
            _, prompt_id = self.q.get()
            top_prompts.append(self.prompt_db[prompt_id])
        return top_prompts
